class LazySingleton
{
public:
	static LazySingleton& get_instance()
	{
		static std::once_flag s_flag;
		std::call_once(s_flag, [&]() {
			instance = new LazySingleton(); }
			);
		return *instance; 
	}

private:
	LazySingleton();
	static LazySingleton* instance;
};
